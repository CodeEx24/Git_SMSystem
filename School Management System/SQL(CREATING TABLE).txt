CREATE TABLE COURSE(
    COURSEID CHAR(8) NOT NULL PRIMARY KEY,
    COURSENAME VARCHAR(100) NOT NULL
);

CREATE TABLE STUDENTS(
    STUDID CHAR(8) NOT NULL,
    LNAME VARCHAR(25) NOT NULL,
    FNAME VARCHAR(25) NOT NULL,
    MNAME VARCHAR(25),
    COURSEID CHAR(8) NOT NULL,
    PHONE CHAR(11),
    EMAIL VARCHAR(50),
    ADDRESS VARCHAR(200),
	
    PRIMARY KEY(STUDID),
    FOREIGN KEY (COURSEID) REFERENCES COURSE (COURSEID)
);

CREATE TABLE SECTION(
    SECTIONNO INT UNSIGNED,
    COURSEID CHAR(8) NOT NULL,
    DAYTIME VARCHAR(20) NOT NULL,
    LOCATION VARCHAR(100) NOT NULL,
    
    PRIMARY KEY(SECTIONNO),
    FOREIGN KEY (COURSEID) REFERENCES COURSE (COURSEID)
);

CREATE TABLE INSTRUCTOR(
    INSTRUCTORID CHAR(8) NOT NULL PRIMARY KEY,
    LNAME VARCHAR(25) NOT NULL,
    FNAME VARCHAR(25) NOT NULL,
    MNAME VARCHAR(25),
    PHONE CHAR(11),
    EMAIL VARCHAR(50)
);

CREATE TABLE ASSIGN(
    INSTRUCTORID CHAR(8) NOT NULL,
    SECTIONNO INT UNSIGNED NOT NULL,
    SUBJECT VARCHAR(100),
	
    PRIMARY KEY(INSTRUCTORID, SECTIONNO),
    FOREIGN KEY (INSTRUCTORID) REFERENCES INSTRUCTOR (INSTRUCTORID),
    FOREIGN KEY (SECTIONNO) REFERENCES SECTION (SECTIONNO)
);
